.after(content) //在匹配元素集合中的每个元素后面插入指定内容，作为兄弟节点
.before(content) //在匹配元素集合中的每个元素前面插入指定内容，作为兄弟节点

.prepend() //向每个匹配的元素内部添加前置内容
.prependTo() //将需要添加的前置内容加到每个匹配的元素

.insertBefore() //在目标元素前面插入集合中每个匹配的元素
.insertAfter() //在目标元素后面插入集合中每个匹配的元素

$(div).empty() //清空内部的html代码，保留div

$(div).remove() //移除div及内部内容

.detach() //删除页面上的节点，但是节点上的数据与事件不会丢失

.clone() //只克隆结构
.clone(true) //结构、事件与数据都克隆

.replaceWith( newContent ) //用提供的内容替换集合中所有匹配的元素并且返回被删除元素的集合

.next() //查找紧邻的后面同辈元素的元素集合
